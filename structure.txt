Maid_deskpet3/
│
├── SakMaienv/                   # Python虚拟环境 (主项目)
│
├── FishEnv/                     # Python虚拟环境 (FishSpeech TTS)
│
├── SakuraMaid_DB.s3db           # 用户数据库(运行后生成)
│
├── session.json                 # 登录会话(记住我/直接登录)
│
├── requirements.txt             # Python依赖列表
│
├── 桌宠开发方案.md               # 开发规划文档
│
├── structure.txt                # 本文件 - 项目结构说明
│
├── Src/                         # 源代码目录
│   │
│   ├── main.py                  # 入口: 登录 → 配置/跳过 → 桌宠
│   │
│   ├── core/                    # 核心模块
│   │   ├── __init__.py
│   │   ├── config.py            # 路径/常量配置
│   │   ├── pet_window.py        # 桌宠主窗口(QWidget)
│   │   ├── animation.py         # 序列帧动画管理器
│   │   ├── behavior.py          # 行为控制器(表情动画, 已规划)
│   │   └── bubble.py            # 气泡对话框渲染
│   │
│   ├── modules/                 # 功能模块
│   │   ├── __init__.py
│   │   ├── database.py          # SQLite数据库 (WAL模式)
│   │   ├── voice.py             # TTS语音合成(本地sherpa-onnx+在线edge-tts)
│   │   ├── speech_recognition.py # 语音识别(sense-voice, PyAudio)
│   │   ├── vision.py            # 视觉理解(llava看图/屏幕分析)
│   │   ├── fish_tts.py          # FishSpeech音色克隆客户端(独立环境)
│   │   ├── screen_capture.py    # 屏幕捕获(pyautogui+PIL)
│   │   ├── action_engine.py     # 动作引擎(llva看图→LLM规划→执行)
│   │   ├── hotkey.py            # 全局快捷键(Ctrl+Shift+V, WinAPI)
│   │   ├── astrbot_client.py    # AstrBot API客户端(主要)
│   │   ├── llm_client.py        # DeepSeek LLM客户端(备用)
│   │   ├── system_monitor.py    # 系统监控(CPU/内存/磁盘)
│   │   ├── virtual_hand.py      # 虚拟手(鼠标键盘模拟)
│   │   ├── config_store.py      # 配置持久化
│   │   └── scheduler.py         # 日程/邮件轮询
│   │
│   ├── ui/                      # 界面模块
│   │   ├── __init__.py
│   │   ├── login_ui.py          # 登录/注册(记住我/直接登录)
│   │   └── config_ui.py         # 配置UI(左侧导航+右侧面板)
│   │                            # 桌宠设置页: 参考图/音色参数
│   │                            # 系统配置页: AstrBot/TTS/LLM/音量
│   │
│   └── plugins/                 # 插件目录(预留)
│
└── resourses/                   # 资源目录
    ├── example/                 # 素材(角色图/视频)
    ├── behaviors/               # 动作序列帧
    │   └── Hello/               # 挥手动作(10帧)
    ├── models/                  # 模型文件
    │   └── vosk/
    │       └── vosk-model-small-cn-0.22/   # Vosk中文语音模型(68MB)
    └── voices/                  # 音频资源
        └── voice_characters_set/ # 声音角色配置
            ├── character_config.json
            └── maid_sounds_cha.wav
